\textbf{Considere el estado puro de un solo qubit descrito por:}
$$
\ket{\psi (\theta, \phi)} = \cos \left(\dfrac{\theta}{2}\right)\ket{0} + e^{i \phi} \sin
\left(\dfrac{\theta}{2}\right) \ket{1}
$$

\textbf{Suponga que en un experimento de tomografía cuántica obtiene los siguientes valores
esperados:}
$$
\braket{X}=\sin(\theta)\cos(\phi), \ \ \ \braket{Y}=\sin(\theta)\sin(\phi), \ \ \
\braket{Z}=\cos{\theta} 
$$ \vspace{.3cm}
