\textbf{Alice y Bob comparten el estado de Bell genérico}\vspace{.2cm}
\begin{align*}
    \ket{\beta_{mn}} = (I \otimes X^m Z^n) \ket{\beta_{00}}, && m,n \in \{0,1\}
\end{align*}

\textbf{donde m y n indican qué operadores de Pauli se aplican al segundo qubit del par
entrelazado:}
\begin{align*}
    m = 1 \Rightarrow X \text{ actúa}, && n=1 \Rightarrow Z \text{ actúa}
\end{align*}

\textbf{Alice desea teletransportar el estado desconocido}:
\begin{align*}
    \ket{\chi} = \alpha \ket{0} + \beta \ket{1}
\end{align*}
